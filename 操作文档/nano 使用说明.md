nano:
	选项:
	  -m: 鼠标模式；
	  -J <number>: 指定列字符高亮；
	  -B: 自动新建备份文件；
	  -l: 显示行号；
	  -E: tab 符号转空格；
	  -Y<word>: 以指定语法高亮；
	  -v: 只读模式；
	  -$: 自动换行
	  -x: 不显示帮助信息
    快捷键:
	  ctrl+g: 帮助
	  ctrl+y: 上一页
	  ctrl+v: 下一页
	  ctrl+w: 往后查找
	  ctrl+q: 往前查找
	  ctrl+k: 剪切当前行
	  ctrl+u: 粘贴
	  ctrl+o: 保存
	  ctrl+x: 退出
	  ctrl+r: 读取文件
	  ctrl+c: 当前位置
	  ctrl+\: 替换
	  ctrl+_: 跳转到指定行
	  ctrl+t: 调用语法检查
	  ctrl+i: 添加 tab 符号
	  ctrl+m: 新行
	  ctrl+h: 往前删除
	  ctrl+d: 往后删除
	  ctrl+s: 保存不提示
	  esc,u: 撤销
	  esc,e: 重做
	  esc,\: 首行
	  esc,/: 尾行
	  esc,a: 添加标记
	  esc,6: 拷贝当前行
	  esc,]: 跳转到匹配的括号
	  esc,q: 跳转到上一个查找位置
	  esc,w: 跳转到下一个查找位置
	  esc,t: 从光标处剪切到文件尾部
	  esc,x: 切换帮助显示隐藏
	  esc,c: 显示光标位置
	  esc,s: 自动换行
	  esc,n: 行号
	  esc,m: 鼠标支持
	  esc,p: 空格显示成点
	  esc,y: 高亮切换
